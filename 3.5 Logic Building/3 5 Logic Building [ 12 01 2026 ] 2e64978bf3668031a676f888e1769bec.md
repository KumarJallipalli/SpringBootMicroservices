public class Patterns {
    public static void main(String[] args) {
        int n = 10;

        // Pattern 1
        for (int i = 0; i < n; i++) {
            System.out.print("*");
        }
        System.out.println();

        // Pattern 2
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                System.out.print("*");
            }
            System.out.println();
        }
    }
}

        for (int i=0; i<n; i++) {
            for (int j=0; j<n; j++) {
                if ( (i==0 && j>0 && j<n-1) || (i==(n-1)/2) || (j==0 && i>0) || (j==n-1 && i>0) ) {
                    System.out.print("* ");
                }
                else {
                    System.out.print("  ");
                }
            }
            System.out.println();
        }
        /**
         * EXPLANATION:
         *      - In the "Actual A" pattern, 
         *      - we need to Remove the corner Stars 
         *      - This is achieved by adding conditions
         *          - when i==0, ( j>0 and j<n-1 ) to ignore the corner stars in 1st row
         *          - when j==0, ( i>0 ) to ignore the corner star in 1st column
         *          - when j==n-1, ( i>0 ) to ignore the corner star in last column
         *      - Hence the Condition -> (i==0&&j>0&&j<n-1 || i==(n-1)/2 || j==0&&i>0 || j==n-1&&i>0)
